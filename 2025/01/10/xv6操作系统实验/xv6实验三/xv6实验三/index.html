/*recursion printing pagetable */
int _vmprint(pagetable_t pagetable, int depth) {
  //there are 2^9 = 512 PTES in a page table
  for(int i = 0; i < 512; i++) {
    pte_t pte = pagetable[i];

    if(pte & PTE_V) {    //if PTE is effective, printing PTE according to the format.
      printf("..");
      for(int j = 0; j < depth; ++j) {
        printf("..");
      }
      printf("%d: pte %p pa %p\n", i, pte, PTE2PA(pte));
      /*if the page is not leaf node, recursion printing child node*/
      if((pte & (PTE_R | PTE_W | PTE_X)) == 0) {
        //this PTE points to a lower-level page table.
        uint64 child = PTE2PA(pte);
        _vmprint((pagetable_t) child, depth + 1);
      }
    }
  }
  return 0;
}

  printf("page table %p\n", pagetable);
  return _vmprint(pagetable, 0);
