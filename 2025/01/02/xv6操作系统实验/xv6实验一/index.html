//user/sleep.c
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"


int main (int argc, char** argv) {
    //create two pipeline : pp2c used parent process to child process communication, pc2p used child  process to  parent process communication.
    int pp2c[2], pc2p[2];
    pipe(pp2c);
    pipe(pc2p);

    if(fork() != 0) {   //parent process
        //parent send a character to the child
        write(pp2c[1], ".", 1);
        close(pp2c[1]);

        //parent read a character from child
        char buf;
        
        read(pc2p[0], &buf, 1);
        printf("%d: received pong\n", getpid());

        //wait child over
        wait(0);

    }else { //child process
        //child process read a character from parent
        char buf;
        read(pp2c[0], &buf, 1);
        printf("%d: received ping\n", getpid());

        //child send a character to parent
        write(pc2p[1], ".", 1);
        close(pc2p[1]);
    }

    //close pipe
    close(pp2c[0]);
    close(pc2p[0]);
    exit(0);
}
