int
uvmcopy(pagetable_t old, pagetable_t new, uint64 sz)
{
  pte_t *pte;
  uint64 pa, i;
  uint flags;
  // char *mem;

  for(i = 0; i < sz; i += PGSIZE){
    if((pte = walk(old, i, 0)) == 0)
      panic("uvmcopy: pte should exist");
    if((*pte & PTE_V) == 0)
      panic("uvmcopy: page not present");
    pa = PTE2PA(*pte);
    flags = PTE_FLAGS(*pte);

    //remove all PTE_W of PTE of father process, and set PTE_COW. Indicates that the page is a cow
    if(*pte & PTE_W) 
        *pte = (*pte & ~ PTE_W) | PTE_COW;
    
    flags = PTE_FLAGS(*pte); // get current father process flag

    //map the physical pages mapped in the parent process to the child process
    if(mappages(new, i, PGSIZE, (uint64)pa, flags) != 0)
      goto err;
    
    lk_krefpage((void*) pa);

    //   kfree(mem);
    //   goto err;
    // }
  }
  return 0;

 err:
  uvmunmap(new, 0, i / PGSIZE, 1);
  return -1;
}
